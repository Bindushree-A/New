import numpy as np
import random
import math

# -------------------------------
# Weil Sequence
# -------------------------------
def weil_sequence(p):
    seq = np.ones(p-1)
    for i in range(1, p):
        val = pow(i, (p-1)//2, p)
        seq[i-1] = -1 if val == p-1 else 1
    return seq

# -------------------------------
# FAST ACF
# -------------------------------
def compute_acf(x):
    X = np.fft.fft(x)
    return np.round(np.fft.ifft(X * np.conj(X)).real).astype(int)

def PSL(x):
    return np.max(np.abs(compute_acf(x)[1:]))

# -------------------------------
# LIGHT MOVES (FAST)
# -------------------------------
def flip_bits(seq, k):
    idx = np.random.choice(len(seq), k, replace=False)
    seq[idx] *= -1
    return seq

def small_block_flip(seq):
    n = len(seq)
    start = random.randint(0, n-1)
    length = random.randint(5, 30)  # smaller block
    seq[start:start+length] *= -1
    return seq

# -------------------------------
# FAST OPTIMIZER
# -------------------------------
def fast_optimize(seq, iterations=8000):

    current = seq.copy()
    current_psl = PSL(current)

    best = current.copy()
    best_psl = current_psl

    temp = 50

    for i in range(iterations):

        new = current.copy()

        move = random.choice(["bit", "multi", "block"])

        if move == "bit":
            new = flip_bits(new, 1)

        elif move == "multi":
            new = flip_bits(new, random.randint(2, 8))

        else:
            new = small_block_flip(new)

        new_psl = PSL(new)
        delta = new_psl - current_psl

        if delta < 0 or random.random() < math.exp(-delta/temp):
            current = new
            current_psl = new_psl

            if new_psl < best_psl:
                best = new
                best_psl = new_psl

        temp *= 0.999

        if i % 2000 == 0:
            print(f"[Iter {i}] Best PSL: {best_psl}")

    return best, best_psl
